int to_seconds(string time) {
    int h, m, s= 0;
    if (sscanf(time.c_str(), "%d:%d:%d", &h, &m, &s) >= 2)
    {
        int seconds = h * 3600 + m * 60 + s;
        return seconds;
    }
    return 0;
}

string to_timeformat(int time) {
    int sec = time % 60;
    time /= 60;
    int min = time % 60;
    time /= 60;
    int hour = time;
    
    string result = "";
    if(hour < 10) result += "0";
    result += to_string(hour) + ":";
    if (min < 10) result += "0";
    result += to_string(min) + ":";
    if (sec < 10) result += "0";
    result += to_string(sec);
    
    return result;
}
